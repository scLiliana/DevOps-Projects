
```bash

REGION="us-east-1" 
IDS_FILE="$HOME/aws_ids.env"

# Función para guardar IDs automáticamente en un documento 
save_id() {
    local VAR_NAME=$1
    local VAR_VALUE=$2
    echo "export $VAR_NAME=\"$VAR_VALUE\"" >> $IDS_FILE
    echo "✅ $VAR_NAME=$VAR_VALUE"
}


echo "Para cargar los IDs en cualquier sesión futura:" 
echo "source $IDS_FILE"
```
