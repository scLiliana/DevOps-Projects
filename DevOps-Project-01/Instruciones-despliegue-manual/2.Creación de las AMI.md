# Creación de las AMI

##  AMI base Linux

Para poder crear las AMI se requiere una base que tenga el sistema operativo que se usará

```bash
# AL2_AMI = AMI ID de Amazon Linux 2 más reciente

AL2_AMI=$(aws ec2 describe-images \
    --owners amazon \
    --filters "Name=name,Values=amzn2-ami-hvm-*-x86_64-gp2" \
              "Name=state,Values=available" \
    --query "sort_by(Images, &CreationDate)[-1].ImageId" \
    --output text \
    --region us-east-1)
    
# Verificación de que se guardó el valor

echo "Amazon Linux 2 AMI: $AL2_AMI"

```


## Global AMI

Global AMI (base para todas las instancias)
**Pre-requisitos**:  ```$AL2_AMI, DEFAULT_SG, DEFAULT_SUBNET```

### Iniciar instancia base
Se lanza una instancia base con la AMI de Linux 2

```bash
# Se lanza la instancia y se guarda su ID en $INSTANCE_BASE

INSTANCE_BASE=$(aws ec2 run-instances \
    --image-id $AL2_AMI \
    --count 1 \
    --instance-type t3.micro \
    --key-name DevOpsKeyPair \
    --security-group-ids $DEFAULT_SG \
    --subnet-id $DEFAULT_SUBNET \
    --associate-public-ip-address \
    --tag-specifications 'ResourceType=instance,Tags=[{Key=Name,Value=GlobalAMI-Base}]' \
    --query "Instances[0].InstanceId" \
    --output text \
    --region us-east-1)
    
# Verificación de que se guardó el valor
echo "Instancia Base: $INSTANCE_BASE"

```

```bash
# Esperar a que la instancia esté lista

aws ec2 wait instance-running \
    --instance-ids $INSTANCE_BASE \
    --region us-east-1
```

### Conexion SSH con la instancia base e instalación de agentes

**Pre-requisitos**:  ```$INSTANCE_BASE```
```bash
# Consultar y guardar la IP pública de la instancia para realizar la conexión

IP_BASE=$(aws ec2 describe-instances \
    --instance-ids $INSTANCE_BASE \
    --query "Reservations[0].Instances[0].PublicIpAddress" \
    --output text \
    --region us-east-1)
    
echo "IP Base: $IP_BASE"
```
**Pre-requisitos**:  ```$IP_BASE```
```bash
# Conectarse e instalar los agentes

ssh -i ~/.ssh/DevOpsKeyPair.pem ec2-user@$IP_BASE << 'EOF'
    sudo yum install -y amazon-cloudwatch-agent
    sudo /opt/aws/amazon-cloudwatch-agent/bin/amazon-cloudwatch-agent-ctl -a start
    sudo yum install -y amazon-ssm-agent
    sudo systemctl start amazon-ssm-agent
    sudo systemctl enable amazon-ssm-agent
    sudo systemctl status amazon-ssm-agent --no-pager
    echo "Agentes instalados correctamente"
EOF

```

### Crear la Global AMI

**Pre-requisitos**:  ```$INSTANCE_BASE```
```bash

GLOBAL_AMI=$(aws ec2 create-image \
    --instance-id $INSTANCE_BASE \
    --name "GlobalAMI-Base" \
    --description "AMI base con CloudWatch y SSM agents" \
    --query "ImageId" \
    --output text \
    --region us-east-1)
    
# Verificación de que se guardó el valor
echo "Global AMI ID: $GLOBAL_AMI  <- GUARDA ESTE ID"

```
**Pre-requisitos**:  ```$GLOBAL_AMI```
```bash
# Esperar a que la AMI esté disponible (~2-5 min)

aws ec2 wait image-available \
    --image-ids $GLOBAL_AMI \
    --region us-east-1
echo "Global AMI lista"

```

```bash
# Terminar la instancia — ya no se necesita

aws ec2 terminate-instances \
    --instance-ids $INSTANCE_BASE \
    --region us-east-1
```



## Ngnix AMI

AMI para el Frontend 
**Pre-requisitos**:  ```$GLOBAL_AMI, DEFAULT_SG, DEFAULT_SUBNET```

### Iniciar instancia base

Se lanza una instancia base con la GlobalAMI-Base

```bash
# Lanzar instancia desde la Global AMI

INSTANCE_NGINX=$(aws ec2 run-instances \
    --image-id $GLOBAL_AMI \
    --count 1 \
    --instance-type t3.micro \
    --key-name DevOpsKeyPair \
    --security-group-ids $DEFAULT_SG \
    --subnet-id $DEFAULT_SUBNET \
    --associate-public-ip-address \
    --tag-specifications 'ResourceType=instance,Tags=[{Key=Name,Value=NginxGoldenAMI-Instance}]' \
    --query "Instances[0].InstanceId" \
    --output text \
    --region us-east-1)
    
# Verificación de que se guardó el valor
echo "Instancia Nginx: $INSTANCE_NGINX"

```

**Pre-requisitos**:  ```$INSTANCE_NGINX```
```bash
# Esperar a que la instancia esté lista

aws ec2 wait instance-running \
    --instance-ids $INSTANCE_NGINX \
    --region us-east-1
```

### Conexion SSH con la instancia base e instalación de agentes

**Pre-requisitos**:  ```$INSTANCE_NGINX```
```bash
# Consultar y guardar la IP pública de la instancia para realizar la conexión

IP_NGINX=$(aws ec2 describe-instances \
    --instance-ids $INSTANCE_NGINX \
    --query "Reservations[0].Instances[0].PublicIpAddress" \
    --output text --region us-east-1)
    
echo "IP Nginx: $IP_NGINX"
```

**Pre-requisitos**:  ```$IP_NGINX```
```bash

# Conectarse e instalar los agentes

# Instalar Nginx
ssh -i ~/.ssh/DevOpsKeyPair.pem ec2-user@$IP_NGINX << 'EOF'
    sudo amazon-linux-extras install nginx1 -y
    sudo systemctl start nginx
    sudo systemctl enable nginx

    sudo tee /usr/local/bin/memory-metrics.sh << 'SCRIPT'
#!/bin/bash
while true; do
    INSTANCE_ID=$(curl -s http://169.254.169.254/latest/meta-data/instance-id)
    MEM_USAGE=$(free | grep Mem | awk '{print $3/$2 * 100.0}')
    aws cloudwatch put-metric-data \
        --metric-name MemoryUsage \
        --namespace Custom \
        --value $MEM_USAGE \
        --dimensions InstanceId=$INSTANCE_ID
    sleep 60
done
SCRIPT
    sudo chmod +x /usr/local/bin/memory-metrics.sh
    nginx -v
    echo "Nginx instalado correctamente"
EOF

```

### Verificar que Nginx está corriendo:

```bash
# Conectarse e inmediatamente verificar el estado
ssh -i ~/.ssh/DevOpsKeyPair.pem ec2-user@$IP_NGINX << 'EOF'
    echo "=== Estado de Nginx ==="
    sudo systemctl status nginx --no-pager

    echo ""
    echo "=== Versión de Nginx ==="
    nginx -v

    echo ""
    echo "=== Nginx habilitado en arranque ==="
    sudo systemctl is-enabled nginx

    echo ""
    echo "=== Respuesta HTTP en localhost ==="
    curl -s -o /dev/null -w "HTTP Status: %{http_code}\n" http://localhost

    echo ""
    echo "=== Script de métricas existe y es ejecutable ==="
    ls -la /usr/local/bin/memory-metrics.sh

    echo ""
    echo "=== Contenido del script ==="
    cat /usr/local/bin/memory-metrics.sh
EOF
```
### Crear Golden AMI de Nginx

**Pre-requisitos**:  ```$INSTANCE_NGINX```


### Conexion SSH con la instancia base e instalación de agentes

**Pre-requisitos**:  ```$INSTANCE_TOMCAT```
```bash
# Consultar y guardar la IP pública de la instancia para realizar la conexión

IP_TOMCAT=$(aws ec2 describe-instances \
    --instance-ids $INSTANCE_TOMCAT \
    --query "Reservations[0].Instances[0].PublicIpAddress" \
    --output text --region us-east-1)
    
echo "IP Tomcat: $IP_TOMCAT"
```

**Pre-requisitos**:  ```$IP_TOMCAT```
```bash

# Conectarse e instalar los agentes

ssh -i ~/.ssh/DevOpsKeyPair.pem  -o StrictHostKeyChecking=no ec2-user@$IP_TOMCAT << 'EOF'
    set -e

    echo "==============================="
    echo "PASO 1: Instalar Java 17 Corretto"
    echo "==============================="
    sudo rpm --import https://yum.corretto.aws/corretto.key
    sudo curl -Lo /etc/yum.repos.d/corretto.repo https://yum.corretto.aws/corretto.repo
    sudo yum install -y java-17-amazon-corretto-devel
    java -version

    echo "==============================="
    echo "PASO 2: Detectar JAVA_HOME"
    echo "==============================="
    JAVA_BIN=$(readlink -f $(which java))
    JAVA_HOME_REAL=$(dirname $(dirname $JAVA_BIN))
    echo "JAVA_HOME detectado: $JAVA_HOME_REAL"

    echo "==============================="
    echo "PASO 3: Descargar Tomcat 9.0.115"
    echo "==============================="
    cd /home/ec2-user
    rm -f apache-tomcat-9.0.115.tar.gz
    wget https://archive.apache.org/dist/tomcat/tomcat-9/v9.0.115/bin/apache-tomcat-9.0.115.tar.gz
    ls -lh apache-tomcat-9.0.115.tar.gz

    # Verificar integridad del tar
    tar -tzf apache-tomcat-9.0.115.tar.gz | head -5

    echo "==============================="
    echo "PASO 4: Extraer e instalar Tomcat"
    echo "==============================="
    sudo tar -xzf apache-tomcat-9.0.115.tar.gz -C /opt/
    sudo rm -f /opt/tomcat
    sudo ln -s /opt/apache-tomcat-9.0.115 /opt/tomcat

    echo "==============================="
    echo "PASO 5: Crear usuario tomcat"
    echo "==============================="
    sudo useradd -r -m -U -d /opt/tomcat -s /bin/false tomcat 2>/dev/null || echo "Usuario ya existe, continuando..."

    echo "==============================="
    echo "PASO 6: Aplicar permisos CORRECTAMENTE"
    echo "==============================="
    # Primero permisos amplios como root, luego ownership
    sudo chmod -R 755 /opt/apache-tomcat-9.0.115
    sudo chown -R tomcat:tomcat /opt/apache-tomcat-9.0.115
    sudo chmod +x /opt/tomcat/bin/*.sh

    # Verificar scripts accesibles
    echo "Scripts disponibles:"
    ls -la /opt/tomcat/bin/*.sh

    echo "==============================="
    echo "PASO 7: Crear servicio systemd"
    echo "==============================="
    sudo tee /etc/systemd/system/tomcat.service > /dev/null << SERVICE
[Unit]
Description=Apache Tomcat Web Application Container
After=network.target

[Service]
Type=forking
Environment=JAVA_HOME=${JAVA_HOME_REAL}
Environment=CATALINA_PID=/opt/tomcat/temp/tomcat.pid
Environment=CATALINA_HOME=/opt/tomcat
Environment=CATALINA_BASE=/opt/tomcat
Environment='CATALINA_OPTS=-Xms512M -Xmx1024M -server -XX:+UseParallelGC'
Environment='JAVA_OPTS=-Djava.awt.headless=true -Djava.security.egd=file:/dev/./urandom'
ExecStart=/opt/tomcat/bin/startup.sh
ExecStop=/opt/tomcat/bin/shutdown.sh
User=tomcat
Group=tomcat
UMask=0007
RestartSec=10
Restart=always

[Install]
WantedBy=multi-user.target
SERVICE

    echo "==============================="
    echo "PASO 8: Arrancar y habilitar Tomcat"
    echo "==============================="
    sudo systemctl daemon-reload
    sudo systemctl enable tomcat
    sudo systemctl start tomcat
    sleep 5
    sudo systemctl status tomcat --no-pager

    echo ""
    echo "✅ Tomcat instalado correctamente"
    echo "JAVA_HOME usado: ${JAVA_HOME_REAL}"
    echo "Tomcat versión: $(ls /opt/ | grep tomcat)"
EOF
```

```bash
    set -e
    
    echo "==============================="
    echo "PASO 1: Instalar Git"
    echo "==============================="
	sudo yum install -y git 
	git --version

    echo "==============================="
    echo "PASO 2: Instalar Java 17 Corretto"
    echo "==============================="
    sudo rpm --import https://yum.corretto.aws/corretto.key
    sudo curl -Lo /etc/yum.repos.d/corretto.repo https://yum.corretto.aws/corretto.repo
    sudo yum install -y java-17-amazon-corretto-devel
    java -version
    
	
    echo "==============================="
    echo "PASO 3: Instalar Maven 3.9.9"
    echo "==============================="
    sudo su
    cd /home/ec2-user
    wget https://archive.apache.org/dist/maven/maven-3/3.9.9/binaries/apache-maven-3.9.9-bin.tar.gz
    ls -lh apache-maven-3.9.9-bin.tar.gz

    sudo tar -xzf apache-maven-3.9.9-bin.tar.gz -C /opt/
    sudo rm -f /opt/maven
    sudo ln -s /opt/apache-maven-3.9.9 /opt/maven

    sudo tee /etc/profile.d/maven.sh > /dev/null << 'PROFILE'
export M2_HOME=/opt/maven
export MAVEN_HOME=/opt/maven
export PATH=$M2_HOME/bin:$PATH
PROFILE

    chmod +x /etc/profile.d/maven.sh
    source /etc/profile.d/maven.sh
    mvn -version

    echo ""
    echo "✅ Maven instalado correctamente"
EOF

```
